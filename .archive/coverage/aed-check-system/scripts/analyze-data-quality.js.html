// Supabase 실제 데이터 품질 분석 스크립트
const { createClient } = require('@supabase/supabase-js');

const supabaseUrl = 'https://bkwpkjtnptbhjqxkfkxa.supabase.co';
const supabaseKey = 'eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9.eyJpc3MiOiJzdXBhYmFzZSIsInJlZiI6ImJrd3BranRucHRiaGpxeGtma3hhIiwicm9sZSI6ImFub24iLCJpYXQiOjE3NTcxNjc1OTksImV4cCI6MjA3Mjc0MzU5OX0.Nq7OPnPcsqKqtkQ4vYO38xnZTMx84R0n_dLTm-Y8tRs';

const supabase = createClient(supabaseUrl, supabaseKey);

async function analyzeDataQuality() {
  console.log('=== AED 데이터 품질 분석 시작 ===\n');

  try {
    // 전체 레코드 수 확인
    const { count: totalCount } = await supabase
      .from('aed_data')
      .select('*', { count: 'exact', head: true });

    console.log(`총 레코드 수: ${totalCount}개\n`);

    // 1. 제조번호 중복 검사
    console.log('1. 제조번호 중복 검사');
    const { data: allSerialNumbers } = await supabase
      .from('aed_data')
      .select('serial_number')
      .not('serial_number', 'is', null);

    const serialNumberCounts = {};
    allSerialNumbers.forEach(item => {
      if (item.serial_number && item.serial_number.trim() !== '') {
        serialNumberCounts[item.serial_number] = (serialNumberCounts[item.serial_number] || 0) + 1;
      }
    });

    const duplicatedSerials = Object.entries(serialNumberCounts)
      .filter(([_, count]) => count > 1);

    console.log(`- 중복된 제조번호 수: ${duplicatedSerials.length}개`);
    if (duplicatedSerials.length > 0) {
      console.log(`- 상위 5개 중복 사례:`);
      duplicatedSerials
        .sort((a, b) => b[1] - a[1])
        .slice(0, 5)
        .forEach(([serial, count]) => {
          console.log(`  * "${serial}": ${count}개 중복`);
        });
    }

    // 제조번호가 비어있는 레코드
    const { count: emptySerialCount } = await supabase
      .from('aed_data')
      .select('*', { count: 'exact', head: true })
      .or('serial_number.is.null,serial_number.eq.""');

    console.log(`- 제조번호 없음: ${emptySerialCount}개\n`);

    // 2. 날짜 논리 오류 검사
    console.log('2. 날짜 논리 오류 검사');

    // 샘플 1000개로 날짜 논리 검사
    const { data: dateCheckSample } = await supabase
      .from('aed_data')
      .select('manufacturing_date, installation_date, first_installation_date')
      .limit(1000);

    let mfgAfterInstall = 0;
    let firstAfterCurrent = 0;
    let futureDate = 0;
    let veryOldDevice = 0;
    const today = new Date();

    dateCheckSample.forEach(item => {
      if (item.manufacturing_date && item.installation_date) {
        const mfgDate = new Date(item.manufacturing_date);
        const installDate = new Date(item.installation_date);

        if (mfgDate > installDate) {
          mfgAfterInstall++;
        }

        if (mfgDate > today) {
          futureDate++;
        }

        const yearsDiff = (today - mfgDate) / (1000 * 60 * 60 * 24 * 365);
        if (yearsDiff > 20) {
          veryOldDevice++;
        }
      }

      if (item.first_installation_date && item.installation_date) {
        const firstDate = new Date(item.first_installation_date);
        const currentDate = new Date(item.installation_date);

        if (firstDate > currentDate) {
          firstAfterCurrent++;
        }
      }
    });

    console.log(`- 제조일 > 설치일: ${mfgAfterInstall}건 (샘플 1000개 중)`);
    console.log(`- 최초설치일 > 현재설치일: ${firstAfterCurrent}건 (샘플 1000개 중)`);
    console.log(`- 미래 날짜 입력: ${futureDate}건 (샘플 1000개 중)`);
    console.log(`- 20년 이상 오래된 장비: ${veryOldDevice}건 (샘플 1000개 중)\n`);

    // 3. GPS 좌표 유효성 검사
    console.log('3. GPS 좌표 유효성 검사');

    // 좌표가 없는 레코드
    const { count: noGPSCount } = await supabase
      .from('aed_data')
      .select('*', { count: 'exact', head: true })
      .or('latitude.is.null,longitude.is.null');

    // 0,0 좌표
    const { count: zeroGPSCount } = await supabase
      .from('aed_data')
      .select('*', { count: 'exact', head: true })
      .eq('latitude', 0)
      .eq('longitude', 0);

    // 한국 범위 벗어난 좌표 (위도 33-39, 경도 124-132)
    const { count: outOfRangeCount } = await supabase
      .from('aed_data')
      .select('*', { count: 'exact', head: true })
      .or('latitude.lt.33,latitude.gt.39,longitude.lt.124,longitude.gt.132');

    console.log(`- GPS 좌표 없음: ${noGPSCount}개`);
    console.log(`- (0,0) 좌표: ${zeroGPSCount}개`);
    console.log(`- 한국 범위 벗어남: ${outOfRangeCount}개\n`);

    // 4. 필수 필드 누락 검사
    console.log('4. 필수 필드 누락 검사');

    const requiredFields = [
      { field: 'management_number', name: '관리번호' },
      { field: 'installation_institution', name: '설치기관' },
      { field: 'installation_position', name: '설치위치' },
      { field: 'model_name', name: '모델명' },
      { field: 'manufacturer', name: '제조사' },
      { field: 'battery_expiry_date', name: '배터리 유효기간' },
      { field: 'patch_expiry_date', name: '패드 유효기간' }
    ];

    for (const { field, name } of requiredFields) {
      const { count } = await supabase
        .from('aed_data')
        .select('*', { count: 'exact', head: true })
        .or(`${field}.is.null,${field}.eq.""`);

      console.log(`- ${name} 누락: ${count}개`);
    }
    console.log('');

    // 5. 운영 상태 일관성 검사
    console.log('5. 운영 상태 일관성 검사');

    // 운영 중인데 외부 미표출이고 사유가 없는 경우
    const { count: noReasonCount } = await supabase
      .from('aed_data')
      .select('*', { count: 'exact', head: true })
      .eq('operation_status', '운영')
      .eq('external_display', 'N')
      .or('external_non_display_reason.is.null,external_non_display_reason.eq.""');

    // 새올 삭제인데 운영 중
    const { count: deletedButActiveCount } = await supabase
      .from('aed_data')
      .select('*', { count: 'exact', head: true })
      .eq('saeum_deletion_status', 'Y')
      .eq('operation_status', '운영');

    console.log(`- 운영 중 + 외부 미표출 + 사유 없음: ${noReasonCount}개`);
    console.log(`- 새올 삭제 + 운영 중: ${deletedButActiveCount}개\n`);

    // 6. 유효기간 만료 검사
    console.log('6. 유효기간 만료 검사');

    const todayStr = new Date().toISOString().split('T')[0];
    const thirtyDaysLater = new Date();
    thirtyDaysLater.setDate(thirtyDaysLater.getDate() + 30);
    const thirtyDaysStr = thirtyDaysLater.toISOString().split('T')[0];

    // 배터리 만료
    const { count: batteryExpiredCount } = await supabase
      .from('aed_data')
      .select('*', { count: 'exact', head: true })
      .lt('battery_expiry_date', todayStr);

    // 배터리 30일 이내 만료
    const { count: batterySoonCount } = await supabase
      .from('aed_data')
      .select('*', { count: 'exact', head: true })
      .gte('battery_expiry_date', todayStr)
      .lt('battery_expiry_date', thirtyDaysStr);

    // 패드 만료
    const { count: padExpiredCount } = await supabase
      .from('aed_data')
      .select('*', { count: 'exact', head: true })
      .lt('patch_expiry_date', todayStr);

    // 패드 30일 이내 만료
    const { count: padSoonCount } = await supabase
      .from('aed_data')
      .select('*', { count: 'exact', head: true })
      .gte('patch_expiry_date', todayStr)
      .lt('patch_expiry_date', thirtyDaysStr);

    console.log(`- 배터리 만료: ${batteryExpiredCount}개`);
    console.log(`- 배터리 30일 이내 만료: ${batterySoonCount}개`);
    console.log(`- 패드 만료: ${padExpiredCount}개`);
    console.log(`- 패드 30일 이내 만료: ${padSoonCount}개\n`);

    // 7. 점검 주기 검사
    console.log('7. 점검 주기 검사');

    const sixMonthsAgo = new Date();
    sixMonthsAgo.setMonth(sixMonthsAgo.getMonth() - 6);
    const threeMonthsAgo = new Date();
    threeMonthsAgo.setMonth(threeMonthsAgo.getMonth() - 3);

    // 6개월 초과 미점검
    const { count: overSixMonthsCount } = await supabase
      .from('aed_data')
      .select('*', { count: 'exact', head: true })
      .lt('last_inspection_date', sixMonthsAgo.toISOString().split('T')[0]);

    // 3-6개월 미점검
    const { count: threeToSixMonthsCount } = await supabase
      .from('aed_data')
      .select('*', { count: 'exact', head: true })
      .gte('last_inspection_date', sixMonthsAgo.toISOString().split('T')[0])
      .lt('last_inspection_date', threeMonthsAgo.toISOString().split('T')[0]);

    // 점검 이력 없음
    const { count: noInspectionCount } = await supabase
      .from('aed_data')
      .select('*', { count: 'exact', head: true })
      .is('last_inspection_date', null);

    console.log(`- 6개월 초과 미점검: ${overSixMonthsCount}개`);
    console.log(`- 3-6개월 미점검: ${threeToSixMonthsCount}개`);
    console.log(`- 점검 이력 없음: ${noInspectionCount}개\n`);

    // 8. 카테고리 분류 검사
    console.log('8. 카테고리 분류 검사');

    // 구비의무기관 카테고리 통계
    const { data: categoryStats } = await supabase
      .from('aed_data')
      .select('category_1')
      .limit(1000);

    const categoryCounts = {};
    categoryStats.forEach(item => {
      if (item.category_1) {
        categoryCounts[item.category_1] = (categoryCounts[item.category_1] || 0) + 1;
      }
    });

    console.log(`- 카테고리 분포 (샘플 1000개):`);
    Object.entries(categoryCounts)
      .sort((a, b) => b[1] - a[1])
      .forEach(([category, count]) => {
        console.log(`  * ${category}: ${count}개`);
      });
    console.log('');

    // 9. 연락처 형식 검사
    console.log('9. 연락처 형식 검사');

    // 연락처 없음
    const { count: noContactCount } = await supabase
      .from('aed_data')
      .select('*', { count: 'exact', head: true })
      .or('institution_contact.is.null,institution_contact.eq.""');

    // 샘플로 형식 검사
    const { data: contactSample } = await supabase
      .from('aed_data')
      .select('institution_contact')
      .not('institution_contact', 'is', null)
      .limit(100);

    const phoneRegex = /^(\d{2,3}-\d{3,4}-\d{4}|\d{9,11})$/;
    let invalidFormatCount = 0;

    contactSample.forEach(item => {
      if (item.institution_contact) {
        const cleaned = item.institution_contact.replace(/[^\d-]/g, '');
        if (!phoneRegex.test(cleaned)) {
          invalidFormatCount++;
        }
      }
    });

    console.log(`- 연락처 없음: ${noContactCount}개`);
    console.log(`- 형식 오류: ${invalidFormatCount}개 (샘플 100개 중)\n`);

    // 10. 제조사/모델 일관성 검사
    console.log('10. 제조사/모델 일관성 검사');

    // 상위 제조사
    const { data: manufacturerData } = await supabase
      .from('aed_data')
      .select('manufacturer')
      .not('manufacturer', 'is', null)
      .limit(1000);

    const manufacturerCounts = {};
    manufacturerData.forEach(item => {
      if (item.manufacturer) {
        manufacturerCounts[item.manufacturer] = (manufacturerCounts[item.manufacturer] || 0) + 1;
      }
    });

    console.log(`- 상위 5개 제조사 (샘플 1000개):`);
    Object.entries(manufacturerCounts)
      .sort((a, b) => b[1] - a[1])
      .slice(0, 5)
      .forEach(([manufacturer, count]) => {
        console.log(`  * ${manufacturer}: ${count}개`);
      });

    // 제조사 없음
    const { count: noManufacturerCount } = await supabase
      .from('aed_data')
      .select('*', { count: 'exact', head: true })
      .or('manufacturer.is.null,manufacturer.eq.""');

    console.log(`- 제조사 정보 없음: ${noManufacturerCount}개\n`);

    // 최종 요약
    console.log('=== 데이터 품질 분석 요약 ===');
    console.log(`총 ${totalCount}개 레코드 중:`);
    console.log(`- 중복 제조번호: ${duplicatedSerials.length}건`);
    console.log(`- 제조번호 없음: ${emptySerialCount}개`);
    console.log(`- GPS 문제: ${noGPSCount + zeroGPSCount + outOfRangeCount}개`);
    console.log(`- 유효기간 만료: ${batteryExpiredCount + padExpiredCount}개`);
    console.log(`- 점검 지연: ${overSixMonthsCount + noInspectionCount}개`);

  } catch (error) {
    console.error('분석 중 오류 발생:', error);
  }
}

// 실행
analyzeDataQuality();
