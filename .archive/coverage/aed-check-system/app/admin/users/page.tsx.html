                              regionCode: selectedUser.region_code ? (REGION_CODE_TO_LABEL[selectedUser.region_code] || selectedUser.region_code) : '',
                              department: selectedUser.department
                            });
                            setDetailEditMode(true);
                          }}
                          className="px-4 py-2 bg-blue-600 hover:bg-blue-700 text-white rounded-md text-sm font-medium transition-colors border border-blue-500 hover:border-blue-600"
                        >
                          정보 수정
                        </button>
                      )}
                      <button
                        onClick={() => {
                          setShowDetailModal(false);
                          setSelectedUser(null);
                        }}
                        className="px-4 py-2 bg-slate-600 hover:bg-slate-700 text-white rounded-md text-sm font-medium transition-colors border border-slate-500 hover:border-slate-600"
                      >
                        닫기
                      </button>
                    </>
                  )}
                </div>
              </div>
            </div>
          </div>
        </div>
      )}
 
      {/* 벌크 승인 패널 */}
      <BulkApprovalPanel
        selectedUsers={Array.from(selectedUsers)}
        users={users}
        onBulkApprove={handleBulkApprove}
        onBulkReject={handleBulkReject}
        onClearSelection={handleClearSelection}
        processing={bulkProcessing}
      />
      </div>
    </NotificationProvider>
  );
}
