'use client';
 
import { useState, useEffect, useRef } from 'react';
import { useRouter } from 'next/navigation';
import { createClient } from '@/lib/supabase/client';
import { GlassCard, NeoButton } from '@/components/ui/modern-mobile';
import { OrganizationAutocomplete } from '@/components/ui/organization-autocomplete';
 
interface ChangeRequest {
  id?: string;
  user_id?: string;
  current_organization?: string;
  current_region?: string;
  new_organization: string;
  new_region: string;
  reason: string;
  status?: 'pending' | 'approved' | 'rejected';
  created_at?: string;
}
 
export default function ChangeOrganizationPage() {
  const router = useRouter();
  const supabase = createClient();
  const [loading, setLoading] = useState(true);
  const [submitting, setSubmitting] = useState(false);
  const [currentProfile, setCurrentProfile] = useState<{
    id: string;
    organization_name?: string;
    region?: string;
    remarks?: string | null;
  } | null>(null);
  const [existingRequest, setExistingRequest] = useState<ChangeRequest | null>(null);
  const [error, setError] = useState<string | null>(null);
  const [success, setSuccess] = useState<string | null>(null);
 
  const [formData, setFormData] = useState({
    newRegion: '',
    newOrganization: '',
    reason: ''
  });
 
  // 타이머 참조 (메모리 누수 방지)
  const navigationTimerRef = useRef<NodeJS.Timeout | null>(null);
 
  // 지역 목록
  const regions = [
    '중앙',
    '서울특별시',
    '부산광역시',
    '대구광역시',
    '인천광역시',
    '광주광역시',
    '대전광역시',
    '울산광역시',
    '세종특별자치시',
    '경기도',
    '강원특별자치도',
    '충청북도',
    '충청남도',
    '전북특별자치도',
    '전라남도',
    '경상북도',
    '경상남도',
    '제주특별자치도'
  ];
 
  useEffect(() => {
    loadProfileAndRequests();
  // eslint-disable-next-line react-hooks/exhaustive-deps
  }, []);
 
  // 컴포넌트 언마운트 시 타이머 정리
  useEffect(() => {
    return () => {
      if (navigationTimerRef.current) {
        clearTimeout(navigationTimerRef.current);
      }
    };
  }, []);
 
  const loadProfileAndRequests = async () => {
    try {
      const { data: { user } } = await supabase.auth.getUser();
 
      if (!user) {
        router.push('/auth/signin');
        return;
      }
 
      // 현재 프로필 로드
      const { data: profile, error: profileError } = await supabase
        .from('user_profiles')
        .select('*')
        .eq('id', user.id)
        .single();
 
      if (profileError) throw profileError;
      setCurrentProfile(profile);
 
      // 기존 변경 요청 확인
      const { data: requests, error: requestError } = await supabase
        .from('organization_change_requests')
        .select('*')
        .eq('user_id', user.id)
        .eq('status', 'pending')
        .single();
 
      if (requests && !requestError) {
        setExistingRequest(requests);
      }
    } catch (err) {
      console.error('Error loading data:', err);
    } finally {
      setLoading(false);
    }
  };
 
  const handleSubmit = async () => {
    if (!formData.newRegion || !formData.newOrganization || !formData.reason) {
      setError('모든 필드를 입력해주세요.');
      return;
    }
 
    if (formData.reason.length < 10) {
      setError('변경 사유를 10자 이상 입력해주세요.');
      return;
    }
 
    setSubmitting(true);
    setError(null);
    setSuccess(null);
 
    try {
      const { data: { user } } = await supabase.auth.getUser();
      if (!user) throw new Error('인증 정보를 찾을 수 없습니다.');
      if (!currentProfile) throw new Error('프로필 정보를 찾을 수 없습니다.');
 
      // organization_change_requests 테이블이 없을 수 있으므로 user_profiles에 임시 저장
      // 실제 운영 시에는 별도 테이블 생성 필요
      const changeRequest: ChangeRequest = {
        user_id: user.id,
        current_organization: currentProfile.organization_name,
        current_region: currentProfile.region,
        new_organization: formData.newOrganization,
        new_region: formData.newRegion,
        reason: formData.reason,
        status: 'pending' as const,
        created_at: new Date().toISOString()
      };
 
      // user_profiles의 remarks 필드에 임시 저장 (실제로는 별도 테이블 필요)
      const existingRemarks = typeof currentProfile.remarks === 'string'
        ? JSON.parse(currentProfile.remarks)
        : currentProfile.remarks || {};
 
      const remarksData = {
        ...existingRemarks,
        organization_change_request: changeRequest
      };
 
      const { error: updateError } = await supabase
        .from('user_profiles')
        .update({
          remarks: JSON.stringify(remarksData),
          updated_at: new Date().toISOString()
        })
        .eq('id', user.id);
 
      if (updateError) throw updateError;
 
      setSuccess('소속 변경 요청이 접수되었습니다. 관리자 승인 후 변경됩니다.');
      setExistingRequest(changeRequest);
 
      // 3초 후 프로필 페이지로 이동
      navigationTimerRef.current = setTimeout(() => {
        router.push('/profile');
      }, 3000);
    // eslint-disable-next-line @typescript-eslint/no-explicit-any
    } catch (err: any) {
      setError(err.message || '요청 처리 중 오류가 발생했습니다.');
    } finally {
      setSubmitting(false);
    }
  };
 
  const cancelRequest = async () => {
    if (!existingRequest || !currentProfile) return;
 
    try {
      // remarks에서 요청 제거
      const existingRemarks = typeof currentProfile.remarks === 'string'
        ? JSON.parse(currentProfile.remarks)
        : currentProfile.remarks || {};
 
      const { error } = await supabase
        .from('user_profiles')
        .update({
          remarks: JSON.stringify({
            ...existingRemarks,
            organization_change_request: null
          }),
          updated_at: new Date().toISOString()
        })
        .eq('id', currentProfile.id);
 
      if (error) throw error;
 
      setExistingRequest(null);
      setSuccess('변경 요청이 취소되었습니다.');
    } catch {
      setError('취소 처리 중 오류가 발생했습니다.');
    }
  };
 
  if (loading) {
    return (
      <div className="min-h-screen bg-gradient-to-br from-gray-950 via-gray-900 to-gray-950 flex items-center justify-center">
        <div className="text-white">로딩 중...</div>
      </div>
    );
  }
 
  return (
    <div className="min-h-screen bg-gradient-to-br from-gray-950 via-gray-900 to-gray-950 p-4">
      <div className="max-w-2xl mx-auto">
        {/* 헤더 */}
        <div className="mb-6 flex items-center justify-between">
          <button
            onClick={() => router.push('/profile')}
            className="text-gray-400 hover:text-white flex items-center gap-2"
          >
            <svg className="w-5 h-5" fill="none" stroke="currentColor" viewBox="0 0 24 24">
              <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M10 19l-7-7m0 0l7-7m-7 7h18" />
            </svg>
            프로필로 돌아가기
          </button>
        </div>
 
        <GlassCard>
          <div className="p-8">
            <h1 className="text-2xl font-bold text-white mb-6">소속 변경 요청</h1>
 
            {/* 알림 메시지 */}
            {error && (
              <div className="mb-6 p-4 bg-red-500/10 border border-red-500/20 rounded-lg">
                <p className="text-red-400">{error}</p>
              </div>
            )}
 
            {success && (
              <div className="mb-6 p-4 bg-green-500/10 border border-green-500/20 rounded-lg">
                <p className="text-green-400">{success}</p>
              </div>
            )}
 
            {/* 기존 요청이 있는 경우 */}
            {existingRequest ? (
              <div className="space-y-6">
                <div className="p-6 bg-yellow-500/10 border border-yellow-500/20 rounded-lg">
                  <h2 className="text-lg font-semibold text-yellow-400 mb-4">
                    변경 요청 대기 중
                  </h2>
                  <div className="space-y-3 text-sm">
                    <div>
                      <span className="text-gray-400">현재 소속:</span>{' '}
                      <span className="text-white">
                        {existingRequest.current_region} - {existingRequest.current_organization}
                      </span>
                    </div>
                    <div>
                      <span className="text-gray-400">변경 요청:</span>{' '}
                      <span className="text-white">
                        {existingRequest.new_region} - {existingRequest.new_organization}
                      </span>
                    </div>
                    <div>
                      <span className="text-gray-400">사유:</span>{' '}
                      <span className="text-white">{existingRequest.reason}</span>
                    </div>
                    <div>
                      <span className="text-gray-400">요청일:</span>{' '}
                      <span className="text-white">
                        {existingRequest.created_at ?
                          new Date(existingRequest.created_at).toLocaleDateString('ko-KR') :
                          '-'
                        }
                      </span>
                    </div>
                  </div>
                  <div className="mt-6">
                    <NeoButton
                      variant="secondary"
                      onClick={cancelRequest}
                    >
                      요청 취소
                    </NeoButton>
                  </div>
                </div>
              </div>
            ) : (
              /* 새 요청 폼 */
              <div className="space-y-6">
                {/* 현재 소속 정보 */}
                <div className="p-4 bg-gray-800/30 rounded-lg">
                  <h2 className="text-sm font-medium text-gray-400 mb-2">현재 소속</h2>
                  <p className="text-white">
                    {currentProfile?.region} - {currentProfile?.organization_name}
                  </p>
                </div>
 
                {/* 변경할 소속 정보 */}
                <div>
                  <h2 className="text-lg font-semibold text-white mb-4">변경할 소속 정보</h2>
 
                  {/* 새 지역 선택 */}
                  <div className="mb-4">
                    <label className="block text-sm text-gray-400 mb-2">새 지역</label>
                    <select
                      value={formData.newRegion}
                      onChange={(e) => setFormData({ ...formData, newRegion: e.target.value, newOrganization: '' })}
                      className="w-full px-4 py-2 bg-gray-800 border border-gray-700 rounded-lg text-white focus:border-green-500 focus:outline-none"
                    >
                      <option value="">지역을 선택하세요</option>
                      {regions.map(region => (
                        <option key={region} value={region}>{region}</option>
                      ))}
                    </select>
                  </div>
 
                  {/* 새 기관 선택 */}
                  <div className="mb-4">
                    <label className="block text-sm text-gray-400 mb-2">새 소속 기관</label>
                    <OrganizationAutocomplete
                      value={formData.newOrganization}
                      onChange={(value) => setFormData({ ...formData, newOrganization: value })}
                      region={formData.newRegion}
                      placeholder={!formData.newRegion ? '지역을 먼저 선택하세요' : '소속 기관을 검색하세요'}
                      disabled={!formData.newRegion}
                    />
                  </div>
 
                  {/* 변경 사유 */}
                  <div>
                    <label className="block text-sm text-gray-400 mb-2">변경 사유</label>
                    <textarea
                      value={formData.reason}
                      onChange={(e) => setFormData({ ...formData, reason: e.target.value })}
                      className="w-full px-4 py-2 bg-gray-800 border border-gray-700 rounded-lg text-white focus:border-green-500 focus:outline-none resize-none"
                      placeholder="소속 변경 사유를 입력해주세요 (최소 10자)"
                      rows={4}
                    />
                  </div>
                </div>
 
                {/* 안내 메시지 */}
                <div className="p-4 bg-blue-500/10 border border-blue-500/20 rounded-lg">
                  <p className="text-blue-400 text-sm">
                    소속 변경은 관리자 승인이 필요합니다. 승인 완료 시 이메일로 안내드립니다.
                  </p>
                </div>
 
                {/* 버튼 영역 */}
                <div className="flex gap-4">
                  <NeoButton
                    variant="secondary"
                    onClick={() => router.push('/profile')}
                    disabled={submitting}
                  >
                    취소
                  </NeoButton>
                  <NeoButton
                    onClick={handleSubmit}
                    disabled={submitting}
                  >
                    {submitting ? '요청 중...' : '변경 요청'}
                  </NeoButton>
                </div>
              </div>
            )}
          </div>
        </GlassCard>
      </div>
    </div>
  );
}
