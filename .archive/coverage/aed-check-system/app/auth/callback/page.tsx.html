'use client';
 
import { useEffect, useState } from 'react';
import { useRouter } from 'next/navigation';
import { createClient } from '@/lib/supabase/client';
 
export default function AuthCallbackPage() {
  const router = useRouter();
  const supabase = createClient();
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);
 
  useEffect(() => {
    const handleCallback = async () => {
      try {
        // 1. 현재 로그인된 사용자 확인
        const { data: { user }, error: authError } = await supabase.auth.getUser();
 
        if (authError || !user) {
          throw new Error('인증 처리 중 오류가 발생했습니다.');
        }
 
        // 2. 이메일 인증 확인
        if (!user.email_confirmed_at) {
          // 이메일 인증이 안된 경우 인증 페이지로
          router.push('/auth/verify-email');
          return;
        }
 
        // 3. sessionStorage에서 저장된 회원가입 정보 가져오기 (우선순위), 없으면 localStorage에서 마이그레이션
        let savedData = sessionStorage.getItem('pendingSignupData');
        if (!savedData) {
          savedData = localStorage.getItem('pendingSignupData');
          if (savedData) {
            // sessionStorage로 마이그레이션
            sessionStorage.setItem('pendingSignupData', savedData);
            localStorage.removeItem('pendingSignupData');
          }
        }
 
        if (savedData) {
          const signupData = JSON.parse(savedData);
 
          // 24시간 이내의 데이터만 유효
          if (Date.now() - signupData.timestamp < 24 * 60 * 60 * 1000) {
            // 4. user_profiles 테이블에 프로필 생성
            const { error: profileError } = await supabase
              .from('user_profiles')
              .insert({
                id: user.id,
                email: user.email || signupData.email,
                full_name: signupData.fullName,
                phone: signupData.phone || null,
                region: signupData.region,
                organization_name: signupData.organizationName,
                remarks: signupData.remarks || null,
                role: 'pending_approval',
                account_type: signupData.accountType || 'public',
                is_active: true
              });
 
            if (profileError) {
              // 이미 프로필이 있는 경우 무시
              if (!profileError.message.includes('duplicate')) {
                console.error('프로필 생성 실패:', profileError);
              }
            }
 
            // 5. sessionStorage 정리
            sessionStorage.removeItem('pendingSignupData');
            sessionStorage.removeItem('pendingVerificationEmail');
            // 기존 localStorage도 정리 (후방 호환성)
            localStorage.removeItem('pendingSignupData');
            localStorage.removeItem('pendingVerificationEmail');
          }
        }
 
        // 5. 프로필 확인
        const { data: profile } = await supabase
          .from('user_profiles')
          .select('*')
          .eq('id', user.id)
          .single();
 
        if (profile) {
          // 프로필이 있으면 역할에 따라 리다이렉트
          if (profile.role === 'pending_approval') {
            router.push('/auth/pending-approval');
          } else {
            router.push('/dashboard');
          }
        } else {
          // 프로필이 없으면 프로필 완성 페이지로
          router.push('/auth/complete-profile');
        }
 
      } catch (err) {
        console.error('콜백 처리 오류:', err);
        setError(err instanceof Error ? err.message : '알 수 없는 오류가 발생했습니다.');
        setLoading(false);
      }
    };
 
    handleCallback();
  }, [supabase, router]);
 
  if (loading) {
    return (
      <div className="min-h-screen bg-black flex items-center justify-center">
        <div className="text-center">
          <div className="animate-spin rounded-full h-12 w-12 border-t-2 border-b-2 border-green-500 mx-auto mb-4"></div>
          <p className="text-white">이메일 확인 처리 중...</p>
        </div>
      </div>
    );
  }
 
  if (error) {
    return (
      <div className="min-h-screen bg-black flex items-center justify-center">
        <div className="text-center">
          <p className="text-red-500 mb-4">{error}</p>
          <button
            onClick={() => router.push('/auth/signin')}
            className="px-4 py-2 bg-green-500 text-white rounded hover:bg-green-600"
          >
            로그인 페이지로
          </button>
        </div>
      </div>
    );
  }
 
  return null;
}
