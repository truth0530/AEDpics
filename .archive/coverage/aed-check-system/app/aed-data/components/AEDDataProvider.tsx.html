'use client';
import { createContext, useContext, useState, useEffect, ReactNode } from 'react';
import useSWR from 'swr';
import { UserProfile } from '@/packages/types';
import { resolveAccessScope, UserAccessScope } from '@/lib/auth/access-control';
import { ParsedFilters, buildQueryString } from '@/lib/utils/query-parser';
import { AEDDevice, AEDDataResponse } from '@/packages/types/aed';
import { ROLE_QUERY_CRITERIA, QueryCriteria } from '@/lib/constants/query-criteria';
import { ToastProvider } from '@/components/ui/Toast';
 
interface AEDDataContextType {
  data?: AEDDevice[];
  isLoading: boolean;
  error?: string;
  summary?: AEDDataResponse['summary'];
  filters: ParsedFilters;
  setFilters: (filters: ParsedFilters) => void;
  pagination?: {
    page: number;
    limit: number;
    hasMore: boolean;
  };
  accessScope?: UserAccessScope;
  filterCapabilities?: AEDDataResponse['filters']['available'];
  userProfile: UserProfile;
  refetch: () => void;
  nextPage: () => void;
  prevPage: () => void;
  resetFilters: () => void;
}
 
const AEDDataContext = createContext<AEDDataContextType | undefined>(undefined);
 
interface AEDDataProviderProps {
  children: ReactNode;
  initialFilters: Record<string, string | string[] | undefined>;
  userProfile: UserProfile;
}
 
async function fetcher(url: string): Promise<AEDDataResponse> {
  const response = await fetch(url);
  if (!response.ok) {
    const error = await response.json();
    throw new Error(error.error || 'Failed to fetch data');
  }
  return response.json();
}
 
export function AEDDataProvider({ children, initialFilters, userProfile }: AEDDataProviderProps) {
  const [accessScope, setAccessScope] = useState<UserAccessScope>();
 
  const getDefaultRegionForEmergencySupport = (): string | undefined => {
    if (
      userProfile.email?.endsWith('@nmc.or.kr') &&
      userProfile.organization?.name?.includes('응급의료지원센터')
    ) {
      const orgName = userProfile.organization.name;
      const regionMap: Record<string, string> = {
        서울: 'SEO',
        부산: 'BUS',
        대구: 'DAE',
        인천: 'INC',
        광주: 'GWA',
        대전: 'DAJ',
        울산: 'ULS',
        세종: 'SEJ',
        경기: 'GYE',
        강원: 'GAN',
        충북: 'CHU',
        충남: 'CHN',
        전북: 'JEO',
        전남: 'JEN',
        경북: 'GYB',
        경남: 'GYN',
        제주: 'JEJ',
      };
      for (const [region, code] of Object.entries(regionMap)) {
        if (orgName.includes(region)) {
          return code;
        }
      }
    }
    return undefined;
  };
 
  const [filters, setFiltersState] = useState<ParsedFilters>(() => {
    const parsed = parseInitialFilters(initialFilters, userProfile.role);
    const defaultRegion = getDefaultRegionForEmergencySupport();
    if ((!parsed.regionCodes || parsed.regionCodes.length === 0) && defaultRegion) {
      parsed.regionCodes = [defaultRegion];
    }
    return {
      page: 1,
      limit: 100,
      ...parsed,
    };
  });
 
  const [accumulatedData, setAccumulatedData] = useState<AEDDevice[]>([]);
 
  useEffect(() => {
    try {
      const scope = resolveAccessScope(userProfile);
      setAccessScope(scope);
    } catch (error) {
      console.error('Failed to resolve access scope:', error);
    }
  }, [userProfile]);
 
  const queryString = buildQueryString(filters);
  const { data: response, error, isLoading, mutate } = useSWR<AEDDataResponse>(
    `/api/aed-data${queryString}`,
    fetcher,
    {
      revalidateOnFocus: false,
      revalidateOnReconnect: false,
    }
  );
 
  useEffect(() => {
    if (response?.data) {
      if ((filters.page || 1) > 1) {
        setAccumulatedData((prev) => [...prev, ...response.data]);
      } else {
        setAccumulatedData(response.data);
      }
    }
  }, [response, filters.page]);
 
  const setFilters = (newFilters: ParsedFilters) => {
    setFiltersState({
      ...newFilters,
      page: 1,
    });
    setAccumulatedData([]);
  };
 
  const nextPage = () => {
    if (response?.pagination.hasMore) {
      setFiltersState((prev) => ({
        ...prev,
        page: (prev.page || 1) + 1,
      }));
    }
  };
 
  const prevPage = () => {
    const currentPage = filters.page || 1;
    if (currentPage > 1) {
      const newPage = currentPage - 1;
      const itemsToKeep = newPage * (filters.limit || 100);
      setAccumulatedData((prev) => prev.slice(0, itemsToKeep));
      setFiltersState((prev) => ({
        ...prev,
        page: newPage,
      }));
    }
  };
 
  const resetFilters = () => {
    const defaultRegion = getDefaultRegionForEmergencySupport();
    const rolePermissions = ROLE_QUERY_CRITERIA[userProfile.role];
    const defaultCriteria = rolePermissions?.defaultCriteria || 'address';
    setFiltersState({
      page: 1,
      limit: 100,
      queryCriteria: defaultCriteria,
      ...(defaultRegion ? { regionCodes: [defaultRegion] } : {}),
    });
    setAccumulatedData([]);
  };
 
  const contextValue: AEDDataContextType = {
    data: accumulatedData,
    isLoading,
    error: error?.message,
    summary: response?.summary,
    filters,
    setFilters,
    pagination: response?.pagination,
    accessScope,
    filterCapabilities: response?.filters.available,
    userProfile,
    refetch: mutate,
    nextPage,
    prevPage,
    resetFilters,
  };
 
  return (
    <ToastProvider>
      <AEDDataContext.Provider value={contextValue}>{children}</AEDDataContext.Provider>
    </ToastProvider>
  );
}
 
export function useAEDData() {
  const context = useContext(AEDDataContext);
  if (context === undefined) {
    throw new Error('useAEDData must be used within an AEDDataProvider');
  }
  return context;
}
 
function parseInitialFilters(
  searchParams: Record<string, string | string[] | undefined>,
  userRole: string,
): Partial<ParsedFilters> {
  const parsed: Partial<ParsedFilters> = {};
  const params = new URLSearchParams();
  Object.entries(searchParams).forEach(([key, value]) => {
    if (Array.isArray(value)) {
      value.forEach((v) => params.append(key, v));
    } else if (value) {
      params.set(key, value);
    }
  });
 
  const expiry = params.get('expiry');
  if (expiry) parsed.expiry = expiry as ParsedFilters['expiry'];
 
  const status = params.getAll('status');
  if (status.length > 0) parsed.status = status as ParsedFilters['status'];
 
  const regions = params.getAll('region');
  if (regions.length > 0) parsed.regionCodes = regions;
 
  const cities = params.getAll('city');
  if (cities.length > 0) parsed.cityCodes = cities;
 
  const criteria = params.get('criteria');
  if (criteria === 'address' || criteria === 'jurisdiction') {
    parsed.queryCriteria = criteria as QueryCriteria;
  } else {
    const rolePermissions = ROLE_QUERY_CRITERIA[userRole];
    parsed.queryCriteria = rolePermissions?.defaultCriteria || 'address';
  }
 
  const page = parseInt(params.get('page') || '1', 10);
  if (!isNaN(page) && page > 0) parsed.page = page;
 
  const limit = parseInt(params.get('limit') || '100', 10);
  if (!isNaN(limit) && limit > 0) parsed.limit = limit;
 
  return parsed;
}
