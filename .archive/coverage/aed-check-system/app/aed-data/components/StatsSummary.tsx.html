'use client';
 
import { useAEDData } from './AEDDataProvider';
import { EXPIRY_FILTER_LABELS } from '@/lib/constants/aed-filters';
 
export function StatsSummary() {
  const { data, summary, filters, isLoading, error, accessScope } = useAEDData();
 
  if (error) {
    return <div className="text-red-400 text-sm">오류: {error}</div>;
  }
 
  if (isLoading) {
    return (
      <div className="animate-pulse">
        <div className="h-6 bg-gray-700 rounded mb-2 w-48" />
        <div className="h-4 bg-gray-700 rounded w-32" />
      </div>
    );
  }
 
  if (!summary) {
    return <div className="text-gray-400 text-sm">데이터를 불러오는 중...</div>;
  }
 
  const totalDevices = summary.totalCount;
  const expiredDevices = summary.expiredCount;
  const expiringSoonDevices = summary.expiringSoonCount;
  const hiddenDevices = summary.hiddenCount;
  const withSensitiveData = summary.withSensitiveDataCount;
 
  return (
    <div className="space-y-4">
      <div>
        <h2 className="text-lg font-semibold text-white mb-1">AED 장비 현황</h2>
        <div className="flex flex-col sm:flex-row sm:items-center sm:justify-between">
          <p className="text-sm text-gray-400">
            {accessScope?.permissions.canViewAllRegions ? '전국' : '관할 지역'} · 총{' '}
            {totalDevices.toLocaleString()}대
          </p>
          {data && (
            <p className="text-xs text-gray-500 mt-1 sm:mt-0">
              현재 페이지: {data.length.toLocaleString()}대 표시 중
            </p>
          )}
        </div>
      </div>
 
      <div className="grid grid-cols-2 lg:grid-cols-4 gap-4">
        <div className="bg-red-900/20 border border-red-800 rounded-lg p-3">
          <div className="text-red-400 text-lg font-bold">{expiredDevices.toLocaleString()}</div>
          <div className="text-red-300 text-xs">만료됨</div>
        </div>
        <div className="bg-yellow-900/20 border border-yellow-800 rounded-lg p-3">
          <div className="text-yellow-400 text-lg font-bold">{expiringSoonDevices.toLocaleString()}</div>
          <div className="text-yellow-300 text-xs">30일 내 만료</div>
        </div>
        <div className="bg-gray-900/50 border border-gray-700 rounded-lg p-3">
          <div className="text-gray-400 text-lg font-bold">{hiddenDevices.toLocaleString()}</div>
          <div className="text-gray-300 text-xs">외부표출차단</div>
        </div>
        {!accessScope?.permissions.canViewSensitiveData && (
          <div className="bg-blue-900/20 border border-blue-800 rounded-lg p-3">
            <div className="text-blue-400 text-lg font-bold">{withSensitiveData.toLocaleString()}</div>
            <div className="text-blue-300 text-xs">민감정보 있음</div>
          </div>
        )}
      </div>
 
      {(filters.expiry || filters.status?.length) && (
        <div className="bg-gray-800/50 border border-gray-700 rounded-lg p-3">
          <div className="text-xs text-gray-400 mb-1">적용된 필터</div>
          <div className="flex flex-wrap gap-2">
            {filters.expiry && (
              <span className="inline-flex items-center px-2 py-1 rounded-full text-xs bg-blue-900 text-blue-300">
                {EXPIRY_FILTER_LABELS[filters.expiry]}
              </span>
            )}
            {filters.status && filters.status.length > 0 && (
              <span className="inline-flex items-center px-2 py-1 rounded-full text-xs bg-green-900 text-green-300">
                상태 {filters.status.length}개
              </span>
            )}
            {filters.regionCodes && filters.regionCodes.length > 0 && (
              <span className="inline-flex items-center px-2 py-1 rounded-full text-xs bg-yellow-900 text-yellow-300">
                지역 {filters.regionCodes.length}개
              </span>
            )}
          </div>
        </div>
      )}
 
      {!accessScope?.permissions.canViewSensitiveData && (
        <div className="text-xs text-yellow-400 bg-yellow-900/10 border border-yellow-800/30 rounded p-2">
          일부 민감정보(연락처, 상세주소)는 마스킹 처리됩니다.
        </div>
      )}
    </div>
  );
}
